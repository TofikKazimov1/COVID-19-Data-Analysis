#Datenbereinigung und -aufbereitung
df["Date"] = pd.to_datetime(df["Date"], dayfirst=True)

df["Date"] = pd.to_datetime(df["Date"])
df["Active"] = df["Confirmed"] - df["Deaths"] - df["Recovered"]
df["Fatality Rate"] = (df["Deaths"] / df["Confirmed"]) * 100
